#include<iostream>
#include<list>

using namespace std;

void display(list<int> &l)
{
	list<int>::iterator p;
	cout<<endl;
	cout<<"Elements in deque : ";
	for(p=l.begin();p!=l.end();++p)
		cout<<*p<<" ";
	cout<<"\n\n";
}

int main()
{
	list<int> l1;

							cout<<"Underflow"<<endl;

						break;
					case 2:
						if(l1.size()!=0)

							cout<<"Underflow"<<endl;

						cout<<"Invalid choice"<<endl;

				break;
			case 3:
				if(l1.size()==0)

				else
					display(l1);
				break;
			default:cout<<"Invalid choice"<<endl;
		}
	cout<<"Enter y to continue =";
	cin>>ch;
	}while(ch=='y'||ch=='Y');
}
