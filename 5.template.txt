#include<iostream>

template<class T>
class vectors

            cin>>m;
            cout<<"Enter new value:";
            cin>>value;

                for(int i=0;i<n;i++)
                {
                    if(i==m-1)
                    {
                        a[i]=value;
                        break;
                    }
                }


            cout<<"{";
        for(int i=0;i<n-1;i++)
        {
            cout<<a[i]<<",";
        }
        cout<<a[n-1]<<"}"<<endl;
            }
            else
            {
                cout<<"Position not found!!\n";

            }
            }
            break;
            case 2:{
                cout<<"Enter value to be replaced:";
            cin>>temp;
            cout<<"Enter new value:";
            cin>>value;
            for(int i=0;i<n;i++)
            {
                if(a[i]==temp)
               {
                    flag=1;
                    a[i]=value;
                    break;
               }
            }
            cout<<"{";
        for(int i=0;i<n-1;i++)
        {
            cout<<a[i]<<",";
        }
        cout<<a[n-1]<<"}"<<endl;
            if(flag==0)
            cout<<"Value not found!!\n";
            }
            break;
            default:cout<<"INVALID "<<endl;
            break;
        }
    }

    void choice()
    {
    	int ch;
    	char c;
    	get_val();
    	cout<<"Vector is = ";
    	put_val();
    do{

        cout<<"Enter your choice:\n1.Multiply given vector by scalar\n2.Modify given vector\n";
        cin>>ch;
        switch(ch)
        {
        case 1:mul_scalar();
                break;
        case 2:modify();
        		break;
        default:cout<<"INVALID CHOICE\n";

        }
        cout<<"Do you want to continue(y/n)??\n";
        cin>>c;
    }while(c=='y'||c=='Y');
}

};
