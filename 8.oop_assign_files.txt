#include<iostream>
#include<fstream>
using namespace std;
int main()
{	
	char ch;
	ofstream o("student.txt");
	cout<<"Enter the text (Enter # to end)\n";
	ch=getchar();
	while(ch!='#')
	{
		o<<ch;
		ch=getchar();
	}
	
	o.close();
	
	ifstream i("student.txt");
	cout<<"\nThis is the text you entered : \n\n";
	while(i)
	{
		ch=i.get();
		cout<<ch;
	}
	i.close();
	return 0;
}

Output:

Enter the text (Enter # to end)
practise makes man perfect
#

This is the text you entered : 

practise makes man perfect
