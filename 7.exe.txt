

#include<iostream>
#include<string.h>
using namespace std;
class exe
{
	int age;
	float income;
	string city;
	char c;
public:

	exe()
{
	age=0;
	income=0;
	city="NA";
	c='n';
}
	void get_val()
	{
		cout<<"Enter your age : ";
		cin>>age;
		cout<<"Enter income : ";
		cin>>income;
		cout<<"Enter city : ";
		cin>>city;
		cout<<"Do you have a four wheeler?(y/n) : ";
		cin>>c;
	}
	void put_val()
	{
		cout<<"Age : "<<age<<endl;
		cout<<"Income : "<<income<<endl;
		cout<<"City : "<<city<<endl;
		cout<<"Owner of a 4 wheeler : "<<c<<endl;
	}


	void check()
	{
		try
		{
		if(age<18||age>55)
				throw(age);
		if(income<50000||age>100000)
				throw(income);
		if(city!="pune"&&city!="chennai"&&city!="mumbai"&&city!="bangalore")
			throw(city);
		if(c=='n'||c=='N')
				throw(c);
		cout<<"\nEligible\n";
		}
		catch(int a)
		{
			cout<<"\nNot eligible due to age exception\n";
		}
		catch(float a)
		{
			cout<<"\nNot eligible due to salary exception\n";
		}
		catch(string a)
		{
			cout<<"\nNot eligible due to city exception\n";
		}
		catch(char a)
		{
			cout<<"\nNot eligible due to vehicle exception\n";
		}
	}
};
int main()
{
	exe e;
	e.get_val();
	e.put_val();
	e.check();
	return 0;
}
