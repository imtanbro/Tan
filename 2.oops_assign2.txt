#include <iostream>
#include<string>

using namespace std;

class student
{
    private:
        string nm,div,bld,dob,ad,lin;
        string c;
        int i,n;
        int roll;
        long con;
    public:
        student()
        {
        	ad="noAddress";
        	lin="none";
        	roll=i=n=0;
        	nm="none";
        	div="none";
        	bld="none";
        	dob="none";
        	con=0;
        	c="none";
        }
        student(string n,string cls,string division,string blood,string db,int rl_no, string address,long contact)
        {
        	nm=n;
        	c=cls;
        	div=division;
        	bld=blood;
        	dob=db;
        	roll=rl_no;
        	ad=address;
        	con=contact;
        }
        student(student &x)
        {
        	nm=x.nm;
            roll=x.roll;
            c=x.c;
        	div=x.div;
        	bld=x.bld;
        	dob=x.dob;
        	roll=x.roll;
        	ad=x.ad;
        	con=x.con;
        }
        void get_data()
        {
        cout<<"\nEnter the name of Student:";
        cin>>nm;
        cout<<"\nEnter the class:";
        cin>>c;
        cout<<"Enter the division:";
        cin>>div;
        cout<<"\nEnter the Roll no:";
        cin>>roll;
        cout<<"\nEnter the address:";
        cin>>ad;
        cout<<"\nEnter the contact no:";
        cin>>con;
        cout<<"\nEnter the DOB:";
        cin>>dob;
        cout<<"\nEnter the blood grp:";
        cin>>bld;
        cout<<"\nEnter the license no:";
        cin>>lin;
        }

        void dis()
        {
            cout<<roll<<"\t\t"<<nm<<"\t\t"<<c<<"\t\t"<<div<<"\t\t"<<dob<<"\t\t"<<con<<"\t\t"<<bld<<"\t\t"<<lin<<"\t\t"<<ad<<"\n";
        }
};

int main()
{
	student s1;
	s1.dis();
	cout<<"\n";
	student s2("geet","SE","B","B+","9/8/1990",2,"Worli",678590534);
	s2.dis();
	cout<<"\n";
	student s3(s2);
	s3.dis();
	cout<<"\n";
    int i,n;
    cout<<"\nEnter the no. of Student:";
    cin>>n;
    student s[n];
    for(i=0;i<n;i++)
    {
        s[i].get_data();
    }
    cout<<"ROLL NO.\tNAME\t\tCLASS\t\tDIV\t\tDOB\t\tCONTACT\t\tBLOOD GRP\t\tLICENSE\t\tADDRESS \n";
    for(i=0;i<n;i++)
    {
        s[i].dis();
    }


    return 0;
}
