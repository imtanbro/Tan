#include<iostream>
#include<list>

using namespace std;

void display(list<int> &l)
{
	list<int>::iterator p;
	cout<<endl;
	cout<<"Elements in queue :";
	for(p=l.begin();p!=l.end();++p)
		cout<<*p<<" ";
		cout<<"\n\n";
}

int main()
{
	list<int> l1;
	char ch;
	int k,s;
	list<int>::iterator p;
	cout<<"\n\nEnter the size of queue =";
	cin>>s;
	do
	{
		cout<<"Menu\n1.Push\n2.Pop\n3.Display"<<endl;
		cin>>k;
		switch(k)
		{
			case 1:
				if(l1.size()!=s)
				{
					cout<<" Enter Element =";
					cin>>k;
					l1.push_back(k);
					cout<<"Element pushed"<<endl;
				}
				else
				{
					cout<<"Overflow"<<endl;
				}
				break;
			case 2:
				if(l1.size()!=0)
				{
					k=l1.front();
					cout<<k<<" is poped"<<endl;
					l1.pop_front();
				}
				else
				{
					cout<<"Underflow"<<endl;
				}
				break;
			case 3:
				if(l1.size()==0)
					cout<<" Empty queue";
				else
					display(l1);
				break;
			default:
				cout<<"Invalid choice"<<endl;
		}
		cout<<"Enter y to continue =";
		cin>>ch;
	}while(ch=='y'||ch=='Y');
	return 0;
}
