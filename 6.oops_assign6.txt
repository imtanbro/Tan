#include <iostream>
using namespace std;

template <class T>

void sort(T a[10],int n)
{
	T k;
	for(int i=0;i<n-1;i++)
	{
		for(int j=0;j<n-1-i;j++)
		{
			if(a[j]>a[j+1])
			{
				k=a[j];
				a[j]=a[j+1];
				a[j+1]=k;
			}
		}
	}
	for(int i=0;i<n;i++)
		cout<<a[i]<<"  ";
}

int main()
{
	int l,m,r;
	char c;
	int a[l];
	float b[m];
	do{
	cout<<"Sort\n\nSelect data type\n1.int\n2.float"<<endl;
	cin>>r;
	switch(r)
	{
	case 1:cout<<"Enter the no of integer elements"<<endl;
			cin>>l;
			cout<<"Enter the elements"<<endl;
			for(int i=0;i<l;i++)
				cin>>a[i];
			sort(a,l);
			break;

	case 2:cout<<"Enter the no of float elements"<<endl;
			cin>>m;
			cout<<"Enter the elements"<<endl;
			for(int i=0;i<m;i++)
				cin>>b[i];
			sort(b,m);
			break;

	default:
		cout<<"Invalid";
	}cout<<"\nDo you want to continue?(y/n) : ";
	cin>>c;
	}while(c=='y'||c=='Y');
	return 0;
}
