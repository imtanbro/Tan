
#include<iostream>
#include<string.h>
using namespace std;
class personal
{
	char name[20];
	string addr;
	char dob[11];
	char gender[12];

public:
	personal()
{
		strcpy(name,"NA");
		addr="NA";
		strcpy(dob,"NA");
		strcpy(gender,"NA");

}
	personal(char n[],string a,char d[],char g[])
{
		strcpy(name,n);
		addr=a;
		strcpy(dob,d);
		strcpy(gender,g);

}
	void get_personal()
	{
		cout<<"Enter your name"<<endl;
		cin>>name;
		cout<<"Enter your address"<<endl;
		cin>>addr;
		cout<<"Enter your date of birth"<<endl;
		cin>>dob;
		cout<<"Enter your gender(Male/Female)"<<endl;
		cin>>gender;
	}
	void put_personal()
	{
		cout<<"Name:"<<name<<endl;
		cout<<"Address:"<<addr<<endl;
		cout<<"Date of birth:"<<dob<<endl;
		cout<<"Gender"<<gender<<endl;

	}
};
	class academic
	{
		char university_passed[50];
		int year_of_passing;
		float per;
		char cls[20];
	public:
		academic()
		{
				strcpy(university_passed,"NA");
				year_of_passing=0000;
				per=0;
				strcpy(cls,"NA");

		}
		academic(char u[],int y,float p,char c[])
		{
				strcpy(university_passed,u);
				year_of_passing=y;
				per=p;
				strcpy(cls,c);

		}
		void get_academic()
	{
		cout<<"Enter the name of university:"<<endl;
		cin>>university_passed;
		cout<<"Enter year of passing:"<<endl;
		cin>>year_of_passing;
		cout<<"Enter your percentage:"<<endl;
		cin>>per;
		cout<<"Enter your passing class:"<<endl;
		cin>>cls;
	}
		void put_academic()
	{
		cout<<"University Name:"<<university_passed<<endl;
		cout<<"Year of passing:"<<year_of_passing<<endl;
		cout<<"Percentage:"<<per<<endl;
		cout<<"Class:"<<cls<<endl;

	}
	};

	class biodata:public personal,public academic,public professional
